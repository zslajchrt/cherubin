# Create DB

DROP TABLE IF EXISTS SOUND;
CREATE TABLE SOUND(
    ID INT PRIMARY KEY AUTO_INCREMENT,
    NAME VARCHAR(255),
    CATEGORY INT,
    SOUNDSET VARCHAR(255),
    SYSEX BLOB
);


DROP TABLE IF EXISTS SYNTH;
CREATE TABLE SYNTH(
    ID INT PRIMARY KEY AUTO_INCREMENT,
    NAME VARCHAR(255),
    TYPE VARCHAR(255)
);

DROP TABLE IF EXISTS SYNTH_SOUND;
CREATE TABLE SYNTH_SOUND(
    ID INT PRIMARY KEY AUTO_INCREMENT,
    BANK INT,
    SLOT INT,
    NAME VARCHAR(255),
    CATEGORY INT,
    SOUNDSET VARCHAR(255),
    SYSEX BLOB,
    SYNTH_ID INT,

    CONSTRAINT FK_SYNTH FOREIGN KEY (SYNTH_ID) REFERENCES SYNTH
)
