# Create DB

DROP TABLE IF EXISTS SOUND;
CREATE TABLE SOUND(
    ID INT PRIMARY KEY AUTO_INCREMENT,
    NAME VARCHAR(255),
    CATEGORY INT,
    SOUNDSET VARCHAR(255),
    SYSEX BLOB
);


DROP TABLE IF EXISTS BLOFELD;
CREATE TABLE BLOFELD(
    ID INT PRIMARY KEY AUTO_INCREMENT,
    NAME VARCHAR(255)
);

INSERT INTO BLOFELD (NAME) VALUES ('INIT')

DROP TABLE IF EXISTS BLOFELD_SINGLE;
CREATE TABLE BLOFELD_SINGLE(
    ID INT PRIMARY KEY AUTO_INCREMENT,
    BANK INT,
    SLOT INT,
    NAME VARCHAR(255),
    CATEGORY INT,
    SOUNDSET VARCHAR(255),
    SYSEX BLOB,
    BLOFELD_ID INT,

    CONSTRAINT FK_BLOFELD FOREIGN KEY (BLOFELD_ID) REFERENCES BLOFELD
)
